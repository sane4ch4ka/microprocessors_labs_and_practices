P2 EQU 0A0H

ORG 0000H
    LJMP START

ORG 0030H

START:
    MOV P2,#080H

MAIN:
    MOV R7,#080H

PAUSE0:
    MOV P2,#080H
    DJNZ R7,PAUSE0

    MOV R0,#000H

LINE1:
    MOV P2,R0
    INC R0
    CJNE R0,#000H,LINE1

    MOV R7,#080H

PAUSE1:
    MOV P2,#080H
    DJNZ R7,PAUSE1

    MOV R0,#080H

LINE2:
    MOV P2,R0
    INC R0
    CJNE R0,#000H,LINE2

    MOV R7,#0A0H

PAUSE2:
    MOV P2,#080H
    DJNZ R7,PAUSE2

    LJMP MAIN

END
